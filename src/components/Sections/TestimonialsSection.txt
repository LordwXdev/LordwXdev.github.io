"use client";
import React, { useState } from 'react';
import { ReviewCard } from '../Cards/ReviewCard';
import { CustomText } from '../UI/CustomText';
import { ChevronLeft, ChevronRight } from 'lucide-react';

const testimonials = [
  {
    name: "Sarah Jenkins",
    title: "CEO at TechFlow",
    rating: 5,
    review: "Working with Lord was a game-changer for our platform. The attention to detail in the UI and the performance of the Next.js site exceeded our expectations."
  },
  {
    name: "Michael Chen",
    title: "Product Manager",
    rating: 5,
    review: "The AI integration Lord developed for our customer portal has reduced support tickets by 40%. Highly professional and technically brilliant."
  },
  {
    name: "Elena Rodriguez",
    title: "Creative Director",
    rating: 4,
    review: "The portfolio Lord built for my agency is stunning. The animations are smooth, and the mobile experience is flawless. Highly recommended!"
  }
];

export const TestimonialsSection = () => {
  const [currentIndex, setCurrentIndex] = useState(0);

  const next = () => {
    setCurrentIndex((prev) => (prev + 1) % testimonials.length);
  };

  const prev = () => {
    setCurrentIndex((prev) => (prev - 1 + testimonials.length) % testimonials.length);
  };

  return (
    <section id="testimonials" className="py-20 bg-white">
      <div className="max-w-7xl mx-auto px-6 text-center">
        
        <h2 className="text-4xl font-extrabold text-gray-900 mb-4">
          <CustomText text="Client Feedback" />
        </h2>
        <p className="text-xl text-gray-600 mb-12">Don't just take my word for it.</p>

        <div className="relative max-w-4xl mx-auto">
          {/* Main Card Display */}
          <div className="overflow-hidden">
            <div 
              className="flex transition-transform duration-500 ease-in-out" 
              style={{ transform: `translateX(-${currentIndex * 100}%)` }}
            >
              {testimonials.map((item, index) => (
                <div key={index} className="w-full flex-shrink-0 px-4">
                  <ReviewCard {...item} />
                </div>
              ))}
            </div>
          </div>

          {/* Navigation Buttons */}
          <button 
            onClick={prev}
            className="absolute top-1/2 -left-4 lg:-left-16 transform -translate-y-1/2 p-2 bg-white rounded-full shadow-lg border border-gray-100 hover:text-[#FF8C00] transition"
          >
            <ChevronLeft size={32} />
          </button>
          <button 
            onClick={next}
            className="absolute top-1/2 -right-4 lg:-right-16 transform -translate-y-1/2 p-2 bg-white rounded-full shadow-lg border border-gray-100 hover:text-[#FF8C00] transition"
          >
            <ChevronRight size={32} />
          </button>

          {/* Pagination Dots */}
          <div className="flex justify-center gap-2 mt-8">
            {testimonials.map((_, i) => (
              <button
                key={i}
                onClick={() => setCurrentIndex(i)}
                className={`w-3 h-3 rounded-full transition-all ${currentIndex === i ? 'bg-[#FF8C00] w-6' : 'bg-gray-300'}`}
              />
            ))}
          </div>
        </div>
      </div>
    </section>
  );
};
